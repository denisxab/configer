def sum_two_number(a: int, b: int) -> int:
    '''
    Сложение двух чисел
    >>> sum_two_number(1, 2)
    3
    >>> sum_two_number(3, 3)
    6
    >>> sum_two_number(-10, -10)
    -20

    '''
